lmm_survey <- read_csv("C:\\Users\\Saheed\\Downloads\\LMM Solutions Survey Results - Sheet1.csv", na = c("", "n/a"))

lmm_survey <- lmm_survey %>% separate(Question, into = c("question", "description"), sep = " - ")

colnames(lmm_survey) <- c("Q#", "Question", "Suggestion", unlist(lapply("R", str_c, 1:13)))



plot_fn <- function(x) {
  lmm_survey %>% 
    pivot_longer(R1:R13, names_to = "Responder", values_to = "Score") %>% 
    group_by(Question, Suggestion) %>% 
    mutate(score_sum = sum(Score, na.rm = TRUE)) %>% 
    ungroup() %>% 
    filter(Question == x) %>%
    ggplot(aes(x    = reorder(word(Suggestion, 1, 5), score_sum), 
               y    = Score, 
               fill = Responder)) + 
    coord_flip() + 
    geom_bar(stat = "identity", position = "stack") + 
    labs(title = x)
}
lapply(lmm_survey %>% distinct(Question) %>% pull(), plot_fn)
