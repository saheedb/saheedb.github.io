What does cumulative overdoses over time look like?

all_drugs %>% 
  ## get ymd from date
  mutate(year_date = ymd(mdy_hms(Date))) %>% 
  ## sort by ymd
  arrange(year_date) %>% 
  ## add a column for cumulative cases

Looks quite linear. Can we fit a model?

od_tbl <-
  ## same pipeline as before
  all_drugs %>% 
  mutate(year_date = ymd(mdy_hms(Date))) %>% 
  arrange(year_date) %>% 
  mutate(total = row_number()) %>% 
  group_by(year_date) %>% 
  slice(n())

## perform linear regression
od_model <- 
  summary(lm(total ~ seq_along(total), od_tble))

## plot
od_tbl %>%
  ggplot(aes(x     = year_date, 
             y     = total, 
             color = total)) + 
  geom_point() + 
  scale_x_date(breaks = "year") + 
  labs(title = "Cumulative overdoses over time in CT",
       x     = "Date",
       y     = "Overdoses") + 
  theme_light() + 
  scale_color_gradient(low = "black", 
                       high = "red") + 
  guides(color = FALSE) + 
  ## add fitting line
  geom_smooth(formula = "y ~ x", method = "lm", na.rm = TRUE) +
  ## overlay R-squared
  annotate("text",
           x = ymd("2018-12-01"),
           y = 4000,
           label = parse(
             text = paste0("R^2: ", round(od_model$r.squared, 3)))
           )

The R-squared value indicates a good linear fit.
